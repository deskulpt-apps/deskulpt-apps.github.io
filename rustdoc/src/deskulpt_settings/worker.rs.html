1//! Worker for processing settings-related tasks.
2
3use anyhow::Result;
4use tauri::{AppHandle, Runtime};
5use tokio::sync::mpsc::{self, UnboundedReceiver};
6
7use crate::settings::{ShortcutAction, Theme};
8use crate::{CanvasImode, SettingsExt};
9
10/// Tasks that the worker can process.
11#[allow(clippy::enum_variant_names)] // TODO: Remove when we have non-change tasks
12#[derive(Debug)]
13pub enum WorkerTask {
14    /// Theme has changed.
15    ///
16    /// The worker will trigger all hooks on theme change.
17    ThemeChanged { old: Theme, new: Theme },
18    /// Canvas interaction mode has changed.
19    ///
20    /// The worker will trigger all hooks on canvas interaction mode change.
21    CanvasImodeChanged { old: CanvasImode, new: CanvasImode },
22    /// Shortcut has changed.
23    ///
24    /// The worker will trigger all hooks on shortcut change.
25    ShortcutChanged {
26        action: ShortcutAction,
27        old: Option<String>,
28        new: Option<String>,
29    },
30}
31
32/// Handle for communicating with the worker.
33pub struct WorkerHandle(mpsc::UnboundedSender<WorkerTask>);
34
35/// The main worker loop.
36async fn worker<R: Runtime>(app_handle: AppHandle<R>, mut rx: UnboundedReceiver<WorkerTask>) {
37    while let Some(task) = rx.recv().await {
38        match task {
39            WorkerTask::ThemeChanged { old, new } => {
40                app_handle.settings().trigger_theme_hooks(&old, &new);
41            },
42            WorkerTask::CanvasImodeChanged { old, new } => {
43                app_handle.settings().trigger_canvas_imode_hooks(&old, &new);
44            },
45            WorkerTask::ShortcutChanged { action, old, new } => {
46                app_handle
47                    .settings()
48                    .trigger_shortcut_hooks(&action, old.as_ref(), new.as_ref());
49            },
50        }
51    }
52}
53
54impl WorkerHandle {
55    /// Create a new [`WorkerHandle`] instance.
56    ///
57    /// This immediately spawns a dedicated worker on Tauri's singleton async
58    /// runtime that listens for incoming [`WorkerTask`]s and processes them
59    /// asynchronously in order.
60    pub fn new<R: Runtime>(app_handle: AppHandle<R>) -> Self {
61        let (tx, rx) = mpsc::unbounded_channel();
62        tauri::async_runtime::spawn(async move {
63            worker(app_handle, rx).await;
64        });
65        Self(tx)
66    }
67
68    /// Instruct the worker to process a task.
69    ///
70    /// This does not block. The task is sent to the worker for asynchronous
71    /// processing and does not wait for completion. An error is returned only
72    /// if task submission fails, but not if task processing fails.
73    pub fn process(&self, task: WorkerTask) -> Result<()> {
74        Ok(self.0.send(task)?)
75    }
76}
