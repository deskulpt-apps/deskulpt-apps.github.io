3use std::path::PathBuf;
4use std::sync::{RwLock, RwLockReadGuard};

6use anyhow::{Result, bail};

10use crate::events::UpdateEvent;
11use crate::settings::{Settings, SettingsPatch, ShortcutAction, Theme};

14#[doc(hidden)]
15type OnThemeChange = Box<dyn Fn(&Theme, &Theme) + Send + Sync>;
16
17#[doc(hidden)]
18type OnShortcutChange =
19    Box<dyn Fn(&ShortcutAction, Option<&String>, Option<&String>) + Send + Sync>;
20
21/// The collection of hooks on settings change.
22#[derive(Default)]
23struct SettingsHooks {
24    /// Hooks triggered on theme change.
25    ///
26    /// See [`SettingsManager::on_theme_change`] for registration.
27    on_theme_change: Vec<OnThemeChange>,
28    /// Hooks triggered on shortcut change.
29    ///
30    /// See [`SettingsManager::on_shortcut_change`] for registration.
31    on_shortcut_change: Vec<OnShortcutChange>,
32}
33
34/// Manager for Deskulpt settings.
35pub struct SettingsManager<R: Runtime> {
36    /// The Tauri app handle.
37    app_handle: AppHandle<R>,
38    /// The Deskulpt settings.
39    settings: RwLock<Settings>,
40    /// The handle for the worker.
41    worker: WorkerHandle,
42    /// The path where settings are persisted on disk.
43    persist_path: PathBuf,
44    /// The collection of hooks on settings change.
45    hooks: RwLock<SettingsHooks>,
46}
47
48impl<R: Runtime> SettingsManager<R> {
49    /// Initialize the [`SettingsManager`].
50    ///
51    /// The settings are loaded from disk. If loading fails (which means
52    /// corrupted settings), default settings are used. A worker is started
53    /// immediately.
54    pub fn new(app_handle: AppHandle<R>) -> Self {
55        let persist_path = app_handle
56            .path()
57            .app_local_data_dir()
58            .unwrap()
59            .join("settings.json");
60
61        let settings = Settings::load(&persist_path).unwrap_or_else(|e| {
62            eprintln!("Failed to load settings: {e:?}");
63            Default::default()
64        });
65
66        let worker = WorkerHandle::new(app_handle.clone());
67
68        Self {
69            app_handle,
70            settings: RwLock::new(settings),
71            hooks: Default::default(),
72            worker,
73            persist_path,
74        }
75    }
76
77    /// Get an immutable reference to the current settings.
78    ///
79    /// The returned guard will hold a read lock on the settings until dropped.
80    /// It is the caller's responsibility to drop the guard as soon as possible
81    /// and never attempt to acquire other locks while holding it to avoid
82    /// deadlocks.
83    pub fn read(&self) -> RwLockReadGuard<'_, Settings> {
84        self.settings.read().unwrap()
85    }
86
87    /// Persist the current settings to disk.
88    pub fn persist(&self) -> Result<()> {
89        let settings = self.settings.read().unwrap();
90        settings.dump(&self.persist_path)?;
91        Ok(())
92    }
93
94    /// Register a hook that will be triggered on theme change.
95    ///
96    /// The two arguments are respectively the old and new themes.
97    pub fn on_theme_change<F>(&self, hook: F)
98    where
99        F: Fn(&Theme, &Theme) + Send + Sync + 'static,
100    {
101        let mut hooks = self.hooks.write().unwrap();
102        hooks.on_theme_change.push(Box::new(hook));
103    }
104
105    /// Trigger all registered theme change hooks.
106    pub(crate) fn trigger_theme_hooks(&self, old: &Theme, new: &Theme) {
107        let hooks = self.hooks.read().unwrap();
108        for hook in &hooks.on_theme_change {
109            hook(old, new);
110        }
111    }
112
113    /// Register a hook that will be triggered on shortcut change.
114    ///
115    /// The first argument is the shortcut action. The second and third
116    /// arguments are respectively the old and new shortcuts for that action.
117    /// `None` means that no shortcut was/is assigned for that action.
118    pub fn on_shortcut_change<F>(&self, hook: F)
119    where
120        F: Fn(&ShortcutAction, Option<&String>, Option<&String>) + Send + Sync + 'static,
121    {
122        let mut hooks = self.hooks.write().unwrap();
123        hooks.on_shortcut_change.push(Box::new(hook));
124    }
125
126    /// Trigger all registered shortcut change hooks.
127    pub(crate) fn trigger_shortcut_hooks(
128        &self,
129        action: &ShortcutAction,
130        old: Option<&String>,
131        new: Option<&String>,
132    ) {
133        let hooks = self.hooks.read().unwrap();
134        for hook in &hooks.on_shortcut_change {
135            hook(action, old, new);
136        }
137    }
138
139    /// Update the settings with a patch generated by a closure.
140    ///
141    /// The closure is given an immutable reference to the current settings and
142    /// must return a [`SettingsPatch`] that describes the changes to be made.
143    /// See its documentation for details on how settings patching works. If any
144    /// actual changes are made, an [`UpdateEvent`] will be emitted with the
145    /// updated settings.
146    ///
147    /// The registered hooks for changed settings will be triggered by the
148    /// worker asynchronously. This is done at best effort, meaning that one
149    /// failure will not block other changes from being submitted. Failure to
150    /// submit one or more changes to the worker will result in an error being
151    /// returned at the end. Failure to trigger the hooks will not result in an
152    /// error, and this method does not wait for the hooks to complete.
153    pub fn update_with<F>(&self, patch: F) -> Result<()>
154    where
155        F: FnOnce(&Settings) -> SettingsPatch,
156    {
157        let mut settings = self.settings.write().unwrap();
158        let patch = patch(&settings);
159
160        let mut tasks = vec![];
161        let mut dirty = false;
162
163        if let Some(theme) = patch.theme
164            && settings.theme != theme
165        {
166            let old_theme = std::mem::replace(&mut settings.theme, theme.clone());
167            tasks.push(WorkerTask::ThemeChanged {
168                old: old_theme,
169                new: theme,
170            });
171            dirty = true;
172        }
173
174        if let Some(shortcuts) = patch.shortcuts {
175            for (action, shortcut) in shortcuts {
176                let old_shortcut = match &shortcut {
177                    Some(shortcut) => settings.shortcuts.insert(action.clone(), shortcut.clone()),
178                    None => settings.shortcuts.remove(&action),
179                };
180                if old_shortcut != shortcut {
181                    tasks.push(WorkerTask::ShortcutChanged {
182                        action,
183                        old: old_shortcut,
184                        new: shortcut,
185                    });
186                    dirty = true;
187                }
188            }
189        }
190
191        if let Some(widgets) = patch.widgets {
192            for (id, patch) in widgets {
193                match patch {
194                    Some(patch) => {
195                        let widget = settings.widgets.entry(id).or_insert_with(|| {
196                            dirty = true;
197                            Default::default()
198                        });
199                        dirty |= widget.apply_patch(patch);
200                    },
201                    None => dirty |= settings.widgets.remove(&id).is_some(),
202                }
203            }
204        }
205
206        if dirty {
207            UpdateEvent(&settings).emit(&self.app_handle)?;
208        }
209
210        // TODO: downgrade write lock to read lock when stable on std or when
211        // switching to parking_lot
212        // We cannot release the lock here; otherwise, a later update may sneak
213        // in and tasks will be submitted and processed out of order, causing a
214        // race condition
215        let mut errors = vec![];
216        for task in tasks {
217            if let Err(e) = self.worker.process(task) {
218                errors.push(e);
219            }
220        }
221        if !errors.is_empty() {
222            let message = errors
223                .into_iter()
224                .map(|e| format!("{e:?}"))
225                .collect::<Vec<_>>()
226                .join("\n\n");
227            bail!("One or more changes failed to be submitted\n\n{message}");
228        }
229
230        Ok(())
231    }
232
233    /// Update the settings with a patch.
234    ///
235    /// This is a wrapper of [`Self::update_with`] that takes a fixed patch.
236    ///
237    /// Tauri command: [`crate::commands::update`].
238    pub fn update(&self, patch: SettingsPatch) -> Result<()> {
239        self.update_with(|_| patch)
240    }
