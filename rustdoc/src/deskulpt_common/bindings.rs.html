1//! Common utilities for declaring Deskulpt bindings.
2
3use std::collections::BTreeMap;
4
5use specta::datatype::{DataType, Function};
6use specta::{NamedType, Type, TypeCollection};
7
8use crate::event::Event;
9
10/// A collection of types, events, and commands to be exposed to the frontend.
11///
12/// A [`Bindings`] should always be constructed via a [`BindingsBuilder`].
13pub struct Bindings {
14    /// A specta type collection.
15    pub types: TypeCollection,
16    /// A mapping from event names to their data types.
17    pub events: BTreeMap<&'static str, DataType>,
18    /// A mapping from plugin names to their commands.
19    pub commands: BTreeMap<&'static str, Vec<Function>>,
20}
21
22/// Builder for a [`Bindings`] instance.
23///
24/// This should never be constructed manually in bindings providers; instead,
25/// configure the build script and use [`configure_bindings_builder!`].
26#[derive(Default)]
27pub struct BindingsBuilder {
28    types: TypeCollection,
29    events: BTreeMap<&'static str, DataType>,
30    commands: BTreeMap<&'static str, fn(&mut TypeCollection) -> Vec<Function>>,
31}
32
33impl BindingsBuilder {
34    /// Register a type in the collection.
35    pub fn typ<T: NamedType>(&mut self) -> &mut Self {
36        self.types.register::<T>();
37        self
38    }
39
40    /// Register an event in the collection.
41    pub fn event<T: Event + Type>(&mut self) -> &mut Self {
42        let dt = T::reference(&mut self.types, &[]).inner;
43        self.events.insert(T::NAME, dt);
44        self
45    }
46
47    /// Register commands in the collection.
48    ///
49    /// The argument should be obtained via the [`collect_commands!`] macro.
50    pub fn commands(
51        &mut self,
52        plugin_name: &'static str,
53        commands: fn(&mut TypeCollection) -> Vec<Function>,
54    ) -> &mut Self {
55        self.commands.insert(plugin_name, commands);
56        self
57    }
58
59    /// Build the [`Bindings`] instance.
60    pub fn build(&mut self) -> Bindings {
61        let commands = self
62            .commands
63            .iter_mut()
64            .map(|(k, f)| (*k, f(&mut self.types)))
65            .collect();
66
67        Bindings {
68            types: self.types.clone(),
69            events: self.events.clone(),
70            commands,
71        }
72    }
73}
74
75/// Used in [`BindingsBuilder::commands`].
76///
77/// <div style="display: none;">
78#[doc(inline)]
79pub use specta::function::collect_functions as collect_commands;

83macro_rules! __configure_bindings_builder {
84    () => {
85        include!(concat!(env!("OUT_DIR"), "/configure_bindings_builder.rs"));
86    };
87}
