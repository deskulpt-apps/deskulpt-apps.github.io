1//! Common utilities for declaring Deskulpt bindings.
2
3use std::collections::BTreeMap;
4
5use specta::datatype::{DataType, Function};
6use specta::{NamedType, Type, TypeCollection};
7
8use crate::event::Event;
9
10/// A collection of types, events, and commands to be exposed to the frontend.
11///
12/// This should never be constructed manually in bindings providers; instead,
13/// configure the build script and use [`build_bindings`].
14pub struct Bindings {
15    /// The namespace these bindings belong to, which should be the plugin name.
16    pub namespace: &'static str,
17    /// The specta type collection.
18    pub types: TypeCollection,
19    /// The mapping from event names to their data types.
20    pub events: BTreeMap<&'static str, DataType>,
21    /// The collection of commands.
22    pub commands: Vec<Function>,
23}
24
25/// Builder for a [`Bindings`] instance.
26pub struct BindingsBuilder {
27    namespace: &'static str,
28    types: TypeCollection,
29    events: BTreeMap<&'static str, DataType>,
30    commands: Option<fn(&mut TypeCollection) -> Vec<Function>>,
31}
32
33impl BindingsBuilder {
34    /// Create a new [`BindingsBuilder`] instance.
35    pub fn new(namespace: &'static str) -> Self {
36        Self {
37            namespace,
38            types: Default::default(),
39            events: Default::default(),
40            commands: Default::default(),
41        }
42    }
43
44    /// Register a type in the collection.
45    pub fn typ<T: NamedType>(&mut self) -> &mut Self {
46        self.types.register::<T>();
47        self
48    }
49
50    /// Register an event in the collection.
51    pub fn event<T: Event + Type>(&mut self) -> &mut Self {
52        let dt = T::reference(&mut self.types, &[]).inner;
53        self.events.insert(T::NAME, dt);
54        self
55    }
56
57    /// Register commands in the collection.
58    ///
59    /// The argument should be obtained via the [`collect_commands!`] macro.
60    pub fn commands(&mut self, commands: fn(&mut TypeCollection) -> Vec<Function>) -> &mut Self {
61        self.commands.replace(commands);
62        self
63    }
64
65    /// Build the [`Bindings`] instance.
66    pub fn build(&mut self) -> Bindings {
67        let commands = match self.commands {
68            Some(f) => f(&mut self.types),
69            None => vec![],
70        };
71
72        Bindings {
73            namespace: self.namespace,
74            types: self.types.clone(),
75            events: self.events.clone(),
76            commands,
77        }
78    }
79}
