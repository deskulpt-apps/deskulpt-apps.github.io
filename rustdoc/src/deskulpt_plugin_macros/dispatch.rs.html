1//! Internals of the `#[dispatch]` macro.
2
3use proc_macro::TokenStream;
4use quote::ToTokens;
5use syn::{FnArg, ItemFn, Pat, PatType, ReturnType, parse_macro_input, parse_quote};
6
7/// Token stream processor for the `#[dispatch]` macro.
8///
9/// This clones the AST of the original function and performs the following
10/// modifications:
11///
12/// - Get the type of the `input` parameter of the function and replace it with
13///   `serde_json::Value`. Not having an `input` parameter panics.
14/// - Get the return type of the function and replace it with
15///   `anyhow::Result<serde_json::Value>`. Not specifying an explicit return
16///   type panics.
17/// - Wrap the original function body in a block that deserializes the input,
18///   calls the original function, serializes the output, and returns it. Note
19///   that the original function must have a return type that the `?` operator
20///   can be applied to.
21pub fn proc_dispatch(_attr: TokenStream, item: TokenStream) -> TokenStream {
22    let mut meth = parse_macro_input!(item as ItemFn);
23
24    let mut input_type = None;
25    for arg in &mut meth.sig.inputs {
26        if let FnArg::Typed(PatType { pat, ty, .. }) = arg
27            && let Pat::Ident(ident) = &**pat
28            && ident.ident == "input"
29        {
30            input_type = Some(ty.clone());
31            *ty = Box::new(parse_quote!(::deskulpt_plugin::serde_json::Value));
32        }
33    }
34    let input_type = input_type.expect("Missing `input` parameter");
35
36    let output_type = if let ReturnType::Type(_, ty) = meth.sig.output {
37        ty
38    } else {
39        panic!("Return type must be specified");
40    };
41    meth.sig.output =
42        parse_quote!(-> ::deskulpt_plugin::anyhow::Result<::deskulpt_plugin::serde_json::Value>);
43
44    let original_body = meth.block.clone();
45    meth.block = Box::new(parse_quote!({
46        let context = format!("Failed to deserialize input: {:?}", input);
47        let input: #input_type = ::deskulpt_plugin::anyhow::Context::context(::deskulpt_plugin::serde_json::from_value(input), context)?;
48        let result: #output_type = #original_body;
49        let result = result?;
50        let output = ::deskulpt_plugin::anyhow::Context::context(::deskulpt_plugin::serde_json::to_value(result), "Failed to serialize output")?;
51        Ok(output)
52    }));
53
54    meth.into_token_stream().into()
55}
