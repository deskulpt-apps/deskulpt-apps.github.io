1use std::fs::File;
2use std::io::BufReader;
3use std::path::Path;
4
5use anyhow::{Context, Result};
6use serde::de::DeserializeOwned;
7
8/// Helper trait for loading manifest files from a directory.
9pub trait LoadManifest: Sized + DeserializeOwned {
10    /// The name of the manifest file to look for in the directory.
11    const FILE_NAME: &'static str;
12
13    /// Load the manifest file from the given directory.
14    ///
15    /// This method does not treat the absence of the manifest file as an error.
16    /// Instead, it returns `Ok(None)` in that case. If the file exists but
17    /// cannot be read or parsed, an error is returned. Otherwise, the parsed
18    /// manifest is returned wrapped in `Ok(Some(...))`.
19    fn load(dir: &Path) -> Result<Option<Self>> {
20        let path = dir.join(Self::FILE_NAME);
21        if !path.exists() {
22            return Ok(None);
23        }
24        let file = File::open(&path)
25            .with_context(|| format!("Failed to open manifest file: {}", path.display()))?;
26        let reader = BufReader::new(file);
27        let config = serde_json::from_reader(reader)
28            .with_context(|| format!("Failed to parse manifest file: {}", path.display()))?;
29        Ok(Some(config))
30    }
31}
