3use serde::Deserialize;
4
5use crate::manifest::LoadManifest;
6
7/// The Deskulpt widget manifest.
8#[derive(Debug, Deserialize)]
9#[serde(rename_all = "camelCase")]
10pub struct WidgetManifest {
11    /// The name of the widget.
12    ///
13    /// This is purely used for display purposes. It does not need to be related
14    /// to the widget directory name, and it does not need to be unique.
15    pub name: String,
16    /// The entry point of the widget.
17    ///
18    /// This is the path to the file that exports the widget component. The path
19    /// should be relative to the widget directory.
20    pub entry: String,
21    /// Whether to ignore the widget.
22    ///
23    /// If set to true, the widget will not be discovered by the application.
24    /// This is useful for temporarily disabling a widget without removing it.
25    #[serde(default, skip_serializing)]
26    pub ignore: bool,
27}
28
29impl LoadManifest for WidgetManifest {
30    const FILE_NAME: &'static str = "deskulpt.widget.json";
31}
