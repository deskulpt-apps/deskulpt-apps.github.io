1//! Setup state of frontend windows.
2
3use std::sync::atomic::{AtomicU8, Ordering};
4
5use bitflags::bitflags;
6use deskulpt_common::window::DeskulptWindow;
7
8bitflags! {
9  /// Flags representing the setup state of frontend windows.
10  struct SetupFlags: u8 {
11    const CANVAS = 1 << 0;
12    const MANAGER = 1 << 1;
13  }
14}
15
16impl From<DeskulptWindow> for SetupFlags {
17    fn from(window: DeskulptWindow) -> Self {
18        match window {
19            DeskulptWindow::Canvas => SetupFlags::CANVAS,
20            DeskulptWindow::Manager => SetupFlags::MANAGER,
21        }
22    }
23}
24
25/// Setup state for frontend windows.
26///
27/// This is essentially an atomic bitmask corresponding to [`SetupFlags`]. Each
28/// set bit means that the corresponding window has completed setup.
29#[derive(Default)]
30pub struct SetupState(AtomicU8);
31
32impl SetupState {
33    /// Mark a window as having completed setup.
34    ///
35    /// Returns `true` if all windows have completed setup after this call.
36    pub fn complete(&self, window: DeskulptWindow) -> bool {
37        let flag: SetupFlags = window.into();
38        // Set the corresponding bit and retrieve the previous state; we only
39        // need AcqRel not SeqCst because we don't care about total ordering
40        let prev = self.0.fetch_or(flag.bits(), Ordering::AcqRel);
41        let current = SetupFlags::from_bits_truncate(prev) | flag;
42        current.is_all()
43    }
44}
