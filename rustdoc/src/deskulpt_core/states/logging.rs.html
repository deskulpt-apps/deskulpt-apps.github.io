1//! State management for logging.
2
3use anyhow::Result;
4use tauri::{App, AppHandle, Manager, Runtime};
5use tracing::Level;
6use tracing_appender::non_blocking::{NonBlockingBuilder, WorkerGuard};
7use tracing_appender::rolling::{RollingFileAppender, Rotation};
8use tracing_subscriber::filter::Targets;
9use tracing_subscriber::fmt::time::UtcTime;
10use tracing_subscriber::layer::SubscriberExt;
11use tracing_subscriber::{Registry, fmt};
12
13use crate::path::PathExt;
14
15/// Maximum number of log files to retain.
16const MAX_LOG_FILES: usize = 10;
17
18/// Managed state for logging.
19struct LoggingState {
20    /// Guard that flushes the tracing worker on drop.
21    _guard: WorkerGuard,
22}
23
24/// Extension trait for operations related to logging.
25pub trait LoggingStateExt<R: Runtime>: Manager<R> + PathExt<R> {
26    /// Initialize state management for logging.
27    ///
28    /// This will set up structured logging in newline-delimited JSON format
29    /// with daily rotation and maximum [`MAX_LOG_FILES`] log files retained.
30    /// A panic hook is also set up to log uncaught panics.
31    fn manage_logging(&self) -> Result<()> {
32        let logs_dir = self.logs_dir()?;
33
34        // TODO: load proper log level from settings and support dynamically
35        // changing at runtime; set to lowest level now for debugging purposes
36        let filter = Targets::new().with_target("deskulpt", Level::TRACE);
37
38        let appender = RollingFileAppender::builder()
39            .rotation(Rotation::DAILY)
40            .max_log_files(MAX_LOG_FILES)
41            .filename_prefix("deskulpt")
42            .filename_suffix("log")
43            .build(logs_dir)?;
44        let (writer, guard) = NonBlockingBuilder::default().finish(appender);
45        let fmt_layer = fmt::layer()
46            .json()
47            .with_ansi(false)
48            .with_file(true)
49            .with_line_number(true)
50            .with_timer(UtcTime::rfc_3339())
51            .flatten_event(true)
52            .with_writer(writer);
53
54        let subscriber = Registry::default().with(filter).with(fmt_layer);
55        tracing::subscriber::set_global_default(subscriber)?;
56
57        let previous_hook = std::panic::take_hook();
58        std::panic::set_hook(Box::new(move |panic_info| {
59            tracing_panic::panic_hook(panic_info);
60            previous_hook(panic_info);
61        }));
62
63        self.manage(LoggingState { _guard: guard });
64        Ok(())
65    }
66}
67
68impl<R: Runtime> LoggingStateExt<R> for App<R> {}
69impl<R: Runtime> LoggingStateExt<R> for AppHandle<R> {}
