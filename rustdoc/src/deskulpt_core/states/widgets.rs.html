18/// Task to render a specific widget.
19struct RenderWidgetTask {
20    /// The widget ID.
21    id: String,
22    /// The entry file relative to the widget directory.
23    entry: String,
24}
25
26/// Managed state for Deskulpt widgets.
27struct WidgetsState {
28    /// The widget catalog.
29    catalog: RwLock<WidgetCatalog>,
30    /// Sender for widget rendering tasks.
31    render_tx: mpsc::UnboundedSender<RenderWidgetTask>,
32}
33
34/// Worker function to process widget rendering tasks.
35async fn render_worker<R: Runtime>(
36    app_handle: AppHandle<R>,
37    mut rx: UnboundedReceiver<RenderWidgetTask>,
38) {
39    while let Some(task) = rx.recv().await {
40        let code = async {
41            let widget_dir = app_handle.widgets_dir()?.join(&task.id);
42            let mut bundler = WidgetBundlerBuilder::new(widget_dir, task.entry).build()?;
43            let code = bundler.bundle().await?;
44            Ok::<_, anyhow::Error>(code)
45        }
46        .await
47        .into();
48        let event = RenderWidgetEvent { id: &task.id, code };
49        if let Err(e) = event.emit_to(&app_handle, DeskulptWindow::Canvas) {
50            eprintln!(
51                "Failed to emit RenderWidgetEvent for widget {}: {e:?}",
52                task.id
53            );
54        };
55    }
56}

58/// Extension trait for operations on Deskulpt widgets.
59pub trait WidgetsStateExt<R: Runtime>: Manager<R> + PathExt<R> + SettingsStateExt<R> {
60    /// Initialize state management for Deskulpt widgets.
61    fn manage_widgets(&self) {
62        let (tx, rx) = mpsc::unbounded_channel::<RenderWidgetTask>();
63        let app_handle = self.app_handle().clone();
64        tauri::async_runtime::spawn(async move {
65            render_worker(app_handle, rx).await;
66        });
67
68        self.manage(WidgetsState {
69            catalog: Default::default(),
70            render_tx: tx,
71        });
72    }
73
74    /// Get an immutable reference to the widget catalog.
75    ///
76    /// The returned reference is behind a lock guard, which should be dropped
77    /// as soon as possible to minimize critical section.
78    fn get_widget_catalog(&self) -> RwLockReadGuard<'_, WidgetCatalog> {
79        let state = self.state::<WidgetsState>().inner();
80        state.catalog.read().unwrap()
81    }
82
83    /// Get a mutable reference to the widget catalog.
84    ///
85    /// The returned reference is behind a lock guard, which should be dropped
86    /// as soon as possible to minimize critical section.
87    fn get_widget_catalog_mut(&self) -> RwLockWriteGuard<'_, WidgetCatalog> {
88        let state = self.state::<WidgetsState>().inner();
89        state.catalog.write().unwrap()
90    }
91
92    /// Reload all widgets.
93    ///
94    /// This function loads a new widget catalog from the widgets directory and
95    /// replaces the existing catalog. It then syncs the settings with the
96    /// updated catalog. If any step fails, an error is returned.
97    fn reload_widgets_all(&self) -> Result<()>
98    where
99        Self: Sized,
100    {
101        let widgets_dir = self.widgets_dir()?;
102        let new_catalog = WidgetCatalog::load(widgets_dir)?;
103
104        let state = self.state::<WidgetsState>();
105        let mut catalog = state.catalog.write().unwrap();
106        *catalog = new_catalog;
107        UpdateWidgetCatalogEvent(&catalog).emit(self)?;
108
109        self.update_settings(|settings| catalog.compute_settings_patch(settings))?;
110        Ok(())
111    }
112
113    /// Render a specific widget by its ID.
114    ///
115    /// If the widget does not exist in the catalog, an error is returned.
116    /// Otherwise, a render task is sent to the render worker. If this fails,
117    /// an error is returned as well. This function does not wait for the
118    /// rendering to complete.
119    fn render_widget(&self, id: String) -> Result<()> {
120        let state = self.state::<WidgetsState>();
121        let catalog = state.catalog.read().unwrap();
122        let config = catalog
123            .0
124            .get(&id)
125            .ok_or_else(|| anyhow!("Widget {id} does not exist in the catalog"))?;
126
127        if let Outcome::Ok(config) = config {
128            state.render_tx.send(RenderWidgetTask {
129                id,
130                entry: config.entry.clone(),
131            })?;
132        }
133        Ok(())
134    }
135
136    /// Render all widgets in the catalog.
137    ///
138    /// This function sends render tasks for all widgets in the catalog to the
139    /// render worker. If any task fails to be sent, an error is returned
140    /// containing all individual errors. This function does not wait for the
141    /// rendering to complete.
142    fn render_widgets_all(&self) -> Result<()> {
143        let state = self.state::<WidgetsState>();
144        let catalog = state.catalog.read().unwrap();
145
146        let mut errors = vec![];
147        for (id, config) in catalog.0.iter() {
148            if let Outcome::Ok(config) = config
149                && let Err(e) = state
150                    .render_tx
151                    .send(RenderWidgetTask {
152                        id: id.clone(),
153                        entry: config.entry.clone(),
154                    })
155                    .with_context(|| format!("Failed to send render task for widget {id}"))
156            {
157                errors.push(e);
158            }
159        }
160
161        if !errors.is_empty() {
162            let message = errors
163                .into_iter()
164                .map(|e| format!("{e:?}"))
165                .collect::<Vec<_>>()
166                .join("\n");
167            bail!(message);
168        }
169
170        Ok(())
171    }
172}
