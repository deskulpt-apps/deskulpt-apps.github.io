1//! State management for application setup.
2
3use std::sync::atomic::{AtomicU8, Ordering};
4
5use bitflags::bitflags;
6use deskulpt_common::window::DeskulptWindow;
7use tauri::{App, AppHandle, Manager, Runtime};
8
9bitflags! {
10  /// Flags representing the setup completion state of different windows.
11  struct Flags: u8 {
12    const CANVAS = 1 << 0;
13    const MANAGER = 1 << 1;
14  }
15}
16
17/// Managed state for application setup.
18#[derive(Default)]
19struct SetupState(
20    /// Bitmask corresponding to [`Flags`].
21    ///
22    /// Each set bit means that the corresponding window has completed setup.
23    AtomicU8,
24);
25
26/// Extension trait for operations related to application setup.
27pub trait SetupStateExt<R: Runtime>: Manager<R> {
28    /// Initialize state management for application setup.
29    fn manage_setup(&self) {
30        self.manage(SetupState::default());
31    }
32
33    /// Mark a window as having completed setup.
34    ///
35    /// Returns `true` if all windows have completed setup after this call.
36    fn complete_setup(&self, window: DeskulptWindow) -> bool {
37        let state = self.state::<SetupState>();
38        let bit = match window {
39            DeskulptWindow::Canvas => Flags::CANVAS,
40            DeskulptWindow::Manager => Flags::MANAGER,
41        };
42
43        // Set the corresponding bit and retrieve the previous state; we only
44        // need AcqRel not SeqCst because we don't care about total ordering
45        let prev = state.0.fetch_or(bit.bits(), Ordering::AcqRel);
46        let current = Flags::from_bits_truncate(prev) | bit;
47        current.is_all()
48    }
49}
50
51impl<R: Runtime> SetupStateExt<R> for App<R> {}
52impl<R: Runtime> SetupStateExt<R> for AppHandle<R> {}
