3use std::sync::Mutex;
4
5use anyhow::Result;
6use deskulpt_common::event::Event;
7use deskulpt_common::window::DeskulptWindow;
8use tauri::menu::MenuItem;
9use tauri::{App, AppHandle, Emitter, Manager, Runtime, WebviewWindow};
10
11use crate::events::ShowToastEvent;
12
13/// Canvas interaction mode.
14#[derive(Clone)]
15enum CanvasImode {
16    /// Sink mode.
17    ///
18    /// The canvas is click-through. Widgets are not interactable. The desktop
19    /// is interactable.
20    Sink,
21    /// Float mode.
22    ///
23    /// The canvas is not click-through. Widgets are interactable. The desktop
24    /// is not interactable.
25    Float,
26}
27
28/// The internal of the managed state for canvas interaction mode.
29struct CanvasImodeStateInner<R: Runtime> {
30    /// The interaction mode of the canvas.
31    mode: CanvasImode,
32    /// The menu item for toggling the canvas interaction mode.
33    menu_item: Option<MenuItem<R>>,
34}
35
36impl<R: Runtime> CanvasImodeStateInner<R> {
37    /// Toggle the interaction mode.
38    ///
39    /// This will change the mode and update the menu item text if it exists.
40    fn toggle(&mut self, canvas: &WebviewWindow<R>) -> Result<()> {
41        // The menu item shows the action that will be performed on click, so it
42        // should be the opposite of the mode
43        let (new_mode, new_text) = match self.mode {
44            CanvasImode::Sink => {
45                canvas.set_ignore_cursor_events(false)?;
46                (CanvasImode::Float, "Sink")
47            },
48            CanvasImode::Float => {
49                canvas.set_ignore_cursor_events(true)?;
50                (CanvasImode::Sink, "Float")
51            },
52        };
53
54        self.mode = new_mode;
55        if let Some(menu_item) = &self.menu_item {
56            menu_item.set_text(new_text)?;
57        }
58        Ok(())
59    }
60}
61
62/// Managed state for canvas interaction mode.
63struct CanvasImodeState<R: Runtime>(Mutex<CanvasImodeStateInner<R>>);
64
65/// Extension trait for operations on canvas interaction mode.
66pub trait CanvasImodeStateExt<R: Runtime>: Manager<R> + Emitter<R> {
67    /// Initialize state management for canvas interaction mode.
68    ///
69    /// The canvas is in sink mode by default.
70    fn manage_canvas_imode(&self) {
71        let inner = CanvasImodeStateInner {
72            mode: CanvasImode::Sink,
73            menu_item: None::<MenuItem<R>>,
74        };
75        self.manage(CanvasImodeState(Mutex::new(inner)));
76    }
77
78    /// Set the menu item for toggling canvas interaction mode.
79    fn set_canvas_imode_menu_item(&self, menu_item: &MenuItem<R>) {
80        let state = self.state::<CanvasImodeState<R>>();
81        let mut state = state.0.lock().unwrap();
82
83        // Cloning works because menu items are behind shared references
84        state.menu_item = Some(menu_item.clone());
85    }
86
87    /// Toggle the interaction mode of the canvas window.
88    ///
89    /// This will show a toast message on the canvas window indicating the new
90    /// interaction mode.
91    fn toggle_canvas_imode(&self) -> Result<()>
92    where
93        Self: Sized,
94    {
95        let canvas = DeskulptWindow::Canvas.webview_window(self)?;
96
97        let state = self.state::<CanvasImodeState<R>>();
98        let mut state = state.0.lock().unwrap();
99        state.toggle(&canvas)?;
100
101        let toast_message = match state.mode {
102            CanvasImode::Float => "Canvas floated.",
103            CanvasImode::Sink => {
104                // Toggled from float to sink, so we try to regain focus to
105                // avoid flickering on the first click; failure to do so is not
106                // critical so we consume the error
107                if let Err(e) = canvas.set_focus() {
108                    eprintln!("Failed to gain focus on canvas: {}", e);
109                }
110                "Canvas sunk."
111            },
112        };
113
114        if let Err(e) =
115            ShowToastEvent::Success(toast_message.to_string()).emit_to(self, DeskulptWindow::Canvas)
116        {
117            eprintln!("Failed to emit ShowToastEvent to canvas: {}", e);
118        }
119
120        Ok(())
121    }
122}

125impl<R: Runtime> CanvasImodeStateExt<R> for AppHandle<R> {}
