1//! State management for the initial render.
2
3use std::sync::Mutex;
4
5use anyhow::Result;
6use deskulpt_common::event::Event;
7use deskulpt_common::window::DeskulptWindow;
8use tauri::{App, AppHandle, Emitter, Manager, Runtime};
9
10use crate::events::RenderWidgetsEvent;
11
12/// Inner structure for [`InitialRenderState`].
13#[derive(Default)]
14struct InitialRenderStateInner {
15    /// Whether the canvas is ready to render widgets.
16    ready: bool,
17    /// Pending event to emit when the canvas is ready.
18    ///
19    /// If the manager window wants to emit a [`RenderWidgetsEvent`] to the
20    /// canvas window when it is not ready to render widgets yet, the message
21    /// will be stored in this field and emitted later when the canvas is ready.
22    pending: Option<RenderWidgetsEvent>,
23}
24
25/// Managed state for the initial render.
26///
27/// The first parameter indicates whether the canvas is ready to render widgets,
28/// i.e., whether the listener for the `render-widgets` event has been set up.
29/// The second parameter is an optional pending payload for that event. In
30/// particular, the manager window will that event to the canvas on startup, but
31/// the canvas may not be ready to receive the event at that point. In this
32/// case, we need to store the payload and emit it later when the canvas is
33/// ready.
34#[derive(Default)]
35struct InitialRenderState(Mutex<InitialRenderStateInner>);
36
37/// Extension trait for operations related to the initial render.
38pub trait InitialRenderStateExt<R: Runtime>: Manager<R> + Emitter<R> {
39    /// Initialize state management for the initial render.
40    fn manage_initial_render(&self) {
41        self.manage(InitialRenderState::default());
42    }
43
44    /// Set the canvas as ready to render widgets.
45    ///
46    /// If there is a pending payload, a [`RenderWidgetsEvent`] will be emitted
47    /// to the canvas with that payload.
48    fn set_render_ready(&self) -> Result<()>
49    where
50        Self: Sized,
51    {
52        let state = self.state::<InitialRenderState>();
53        let mut initial_render = state.0.lock().unwrap();
54        initial_render.ready = true;
55
56        if let Some(event) = initial_render.pending.take() {
57            event.emit_to(self, DeskulptWindow::Canvas)?;
58        }
59        Ok(())
60    }
61
62    /// Emit a [`RenderWidgetsEvent`] to the canvas when it is ready.
63    ///
64    /// If the canvas is already ready to render widgets, emit the given payload
65    /// to the canvas immediately. Otherwise, store the payload as pending so
66    /// that it can be emitted later when the canvas is ready.
67    fn emit_on_render_ready(&self, event: RenderWidgetsEvent) -> Result<()>
68    where
69        Self: Sized,
70    {
71        let state = self.state::<InitialRenderState>();
72        let mut initial_render = state.0.lock().unwrap();
73
74        if !initial_render.ready {
75            initial_render.pending = Some(event);
76            return Ok(());
77        }
78        event.emit_to(self, DeskulptWindow::Canvas)?;
79        Ok(())
80    }
81}
82
83impl<R: Runtime> InitialRenderStateExt<R> for App<R> {}
84impl<R: Runtime> InitialRenderStateExt<R> for AppHandle<R> {}
