14/// Extention trait for window-related operations.
15pub trait WindowExt<R: Runtime>: Manager<R> + SettingsStateExt<R> {
16    /// Create the manager window.
17    fn create_manager(&self) -> Result<()>
18    where
19        Self: Sized,
20    {
21        let settings = self.get_settings();
22        let init_js = ManagerInitJS::generate(&settings)?;
23
24        // https://www.radix-ui.com/colors: "Slate 1" colors
25        let background_color = match settings.theme {
26            Theme::Light => (252, 252, 253), // #FCFCFD
27            Theme::Dark => (17, 17, 19),     // #111113
28        };
29
30        WebviewWindowBuilder::new(
31            self,
32            DeskulptWindow::Manager,
33            WebviewUrl::App("packages/deskulpt-manager/index.html".into()),
34        )
35        .title("Deskulpt Manager")
36        .background_color(background_color.into())
37        .inner_size(800.0, 500.0)
38        .center()
39        .resizable(false)
40        .maximizable(false)
41        .minimizable(false)
42        .visible(false)
43        .initialization_script(&init_js)
44        .build()?;
45
46        Ok(())
47    }
48
49    /// Create the canvas window.
50    fn create_canvas(&self) -> Result<()>
51    where
52        Self: Sized,
53    {
54        let settings = self.get_settings();
55        let init_js = CanvasInitJS::generate(&settings)?;
56        let canvas = WebviewWindowBuilder::new(
57            self,
58            DeskulptWindow::Canvas,
59            WebviewUrl::App("packages/deskulpt-canvas/index.html".into()),
60        )
61        .title("Deskulpt Canvas")
62        .maximized(true)
63        .transparent(true)
64        .decorations(false)
65        .always_on_bottom(true)
66        // TODO: Remove when the following issue is fixed:
67        // https://github.com/tauri-apps/tauri/issues/9597
68        .visible(false)
69        // Unsupported on macOS; see below for activation policy
70        .skip_taskbar(true)
71        .initialization_script(&init_js)
72        .build()?;
73
74        #[cfg(target_os = "macos")]
75        {
76            use objc2::msg_send;
77            use objc2::runtime::{AnyObject, Bool};
78
79            // Disable the window shadow on macOS; there will be shadows left on
80            // movement for transparent and undecorated windows that we are using;
81            // it seems that disabling shadows does not have significant visual impacts
82            unsafe {
83                let ns_window = canvas.ns_window()? as *mut AnyObject;
84                let () = msg_send![ns_window, setHasShadow:Bool::NO];
85            }
86        }
87
88        // TODO: Remove when the following issue is fixed:
89        // https://github.com/tauri-apps/tauri/issues/9597
90        canvas.show()?;
91
92        // Canvas is by default click-through
93        canvas.set_ignore_cursor_events(true)?;
94
95        Ok(())
96    }
97
98    /// Open the manager window.
99    fn open_manager(&self) -> Result<()> {
100        let manager = DeskulptWindow::Manager.webview_window(self)?;
101        manager.show()?;
102        manager.set_focus()?;
103        Ok(())
104    }
105}

110/// Window event handler for all Deskulpt windows.
111pub fn on_window_event(window: &Window, event: &WindowEvent) {
112    if window.label() == "manager" {
113        // Prevent the manager window from closing when the close button is
114        // clicked, but only hide it instead
115        if let WindowEvent::CloseRequested { api, .. } = event {
116            api.prevent_close();
117            if let Err(e) = window.hide() {
118                eprintln!("Failed to hide the manager window: {e}");
119            }
120        }
121    }
122}
