17/// Builder for the Deskulpt widget bundler.
18pub struct WidgetBundlerBuilder {
19    /// Absolute path to the widget directory.
20    root: PathBuf,
21    /// Entry file relative to the widget directory.
22    entry: String,
23}
24
25impl WidgetBundlerBuilder {
26    /// Create a new widget bundler builder instance.
27    pub fn new(root: PathBuf, entry: String) -> Self {
28        Self { root, entry }
29    }
30
31    /// Build the Deskulpt widget bundler.
32    pub fn build(self) -> Result<WidgetBundler> {
33        const JSX_RUNTIME_URL: &str = "__DESKULPT_BASE_URL__/gen/jsx-runtime.js";
34        const RAW_APIS_URL: &str = "__DESKULPT_BASE_URL__/gen/raw-apis.js";
35        const REACT_URL: &str = "__DESKULPT_BASE_URL__/gen/react.js";
36        const UI_URL: &str = "__DESKULPT_BASE_URL__/gen/ui.js";
37        const APIS_BLOB_URL: &str = "__DESKULPT_APIS_BLOB_URL__";
38
39        let bundler_options = BundlerOptions {
40            input: Some(vec![self.entry.into()]),
41            cwd: Some(self.root),
42            format: Some(OutputFormat::Esm),
43            platform: Some(Platform::Browser),
44            minify: Some(RawMinifyOptions::Bool(true)),
45            // Use automatic runtime for JSX transforms, which will refer to
46            // `@deskulpt-test/emotion/jsx-runtime`
47            transform: Some(BundlerTransformOptions {
48                jsx: Some(Either::Right(JsxOptions {
49                    runtime: Some("automatic".to_string()),
50                    import_source: Some("@deskulpt-test/emotion".to_string()),
51                    ..Default::default()
52                })),
53                ..Default::default()
54            }),
55            // Externalize default dependencies available at runtime
56            external: Some(
57                vec![
58                    JSX_RUNTIME_URL.to_string(),
59                    RAW_APIS_URL.to_string(),
60                    REACT_URL.to_string(),
61                    UI_URL.to_string(),
62                    APIS_BLOB_URL.to_string(),
63                ]
64                .into(),
65            ),
66            ..Default::default()
67        };
68
69        // Alias the default dependencies to URLs resolvable at runtime
70        let alias_plugin = AliasPlugin(
71            [
72                (
73                    "@deskulpt-test/emotion/jsx-runtime".to_string(),
74                    JSX_RUNTIME_URL.to_string(),
75                ),
76                (
77                    "@deskulpt-test/raw-apis".to_string(),
78                    RAW_APIS_URL.to_string(),
79                ),
80                ("@deskulpt-test/react".to_string(), REACT_URL.to_string()),
81                ("@deskulpt-test/ui".to_string(), UI_URL.to_string()),
82                ("@deskulpt-test/apis".to_string(), APIS_BLOB_URL.to_string()),
83            ]
84            .into(),
85        );
86
87        let bundler = Bundler::with_plugins(bundler_options, vec![Arc::new(alias_plugin)])?;
88        Ok(WidgetBundler { bundler })
89    }
90}
91
92/// The Deskulpt widget bundler.
93pub struct WidgetBundler {
94    bundler: Bundler,
95}
96
97impl WidgetBundler {
98    /// Bundle the widget into a single ESM code string.
99    pub async fn bundle(&mut self) -> Result<String> {
100        let result = self.bundler.generate().await.map_err(|e| {
101            anyhow!(
102                e.into_vec()
103                    .iter()
104                    .map(|diagnostic| diagnostic.to_diagnostic().to_string())
105                    .collect::<Vec<String>>()
106                    .join("\n")
107            )
108        })?;
109
110        // We have supplied a single entry file, so we expect a single output
111        // bundle; this can be broken if widget code contains e.g. dynamic
112        // imports, which we do not allow
113        if result.assets.len() != 1 {
114            bail!(
115                "Expected 1 bundled output, found {}; ensure that widget code does not contain \
116                 e.g. dynamic imports that may result in extra chunks",
117                result.assets.len()
118            );
119        }
120
121        let output = &result.assets[0];
122        let code = match output {
123            Output::Asset(asset) => asset.source.clone().try_into_string()?,
124            Output::Chunk(chunk) => chunk.code.clone(),
125        };
126        Ok(code)
127    }
