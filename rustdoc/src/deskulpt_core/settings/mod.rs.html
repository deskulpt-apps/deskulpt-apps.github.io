34/// Per-widget settings.
35///
36/// Different from widget configurations, these are independent of the widget
37/// configuration files and are managed internally by the application.
38#[serde_as]
39#[derive(Debug, Deserialize, Serialize, JsonSchema, specta::Type)]
40#[serde(rename_all = "camelCase", default)]
41pub struct WidgetSettings {
42    /// The leftmost x-coordinate in pixels.
43    #[serde_as(deserialize_as = "DefaultOnError")]
44    pub x: i32,
45    /// The topmost y-coordinate in pixels.
46    #[serde_as(deserialize_as = "DefaultOnError")]
47    pub y: i32,
48    /// The width in pixels.
49    #[serde_as(deserialize_as = "DefaultOnError")]
50    pub width: u32,
51    /// The height in pixels.
52    #[serde_as(deserialize_as = "DefaultOnError")]
53    pub height: u32,
54    /// The opacity in percentage.
55    #[serde(deserialize_with = "WidgetSettings::deserialize_opacity")]
56    #[schemars(range(min = 1, max = 100))]
57    pub opacity: u8,
58}
59
60impl Default for WidgetSettings {
61    fn default() -> Self {
62        Self {
63            x: 0,
64            y: 0,
65            width: 300,
66            height: 200,
67            opacity: 100,
68        }
69    }
70}

72impl WidgetSettings {
73    /// Deserialization helper for opacity.
74    ///
75    /// On error deserializing this field, it will be set to default (100). The
76    /// deserialized value will be clamped to [1, 100].
77    fn deserialize_opacity<'de, D>(deserializer: D) -> Result<u8, D::Error>
78    where
79        D: Deserializer<'de>,
80    {
81        match u8::deserialize(deserializer) {
82            Ok(opacity) => Ok(opacity.clamp(1, 100)),
83            Err(_) => Ok(100),
84        }
85    }
86}

88/// A patch for partial updates to [`WidgetSettings`].
89#[derive(Debug, Default, Deserialize, specta::Type)]
90#[serde(rename_all = "camelCase", default)]
91pub struct WidgetSettingsPatch {
92    /// If not `None`, update [`WidgetSettings::x`].
93    #[specta(optional, type = i32)]
94    pub x: Option<i32>,
95    /// If not `None`, update [`WidgetSettings::y`].
96    #[specta(optional, type = i32)]
97    pub y: Option<i32>,
98    /// If not `None`, update [`WidgetSettings::width`].
99    #[specta(optional, type = u32)]
100    pub width: Option<u32>,
101    /// If not `None`, update [`WidgetSettings::height`].
102    #[specta(optional, type = u32)]
103    pub height: Option<u32>,
104    /// If not `None`, update [`WidgetSettings::opacity`].
105    #[specta(optional, type = u8)]
106    pub opacity: Option<u8>,
107}
108
109impl WidgetSettings {
110    /// Apply a [`WidgetSettingsPatch`].
111    ///
112    /// This method also returns whether the widget settings is actually changed
113    /// by the patch.
114    pub fn apply_patch(&mut self, patch: WidgetSettingsPatch) -> bool {
115        #[inline]
116        fn set_if_changed<T: PartialEq>(dst: &mut T, src: Option<T>) -> bool {
117            match src {
118                Some(v) if *dst != v => {
119                    *dst = v;
120                    true
121                },
122                _ => false,
123            }
124        }
125
126        let mut dirty = false;
127        dirty |= set_if_changed(&mut self.x, patch.x);
128        dirty |= set_if_changed(&mut self.y, patch.y);
129        dirty |= set_if_changed(&mut self.width, patch.width);
130        dirty |= set_if_changed(&mut self.height, patch.height);
131        dirty |= set_if_changed(&mut self.opacity, patch.opacity);
132        dirty
133    }
134}

136/// Full settings of the Deskulpt application.
137#[serde_as]
138#[derive(Debug, Default, Deserialize, Serialize, JsonSchema, specta::Type)]
139#[serde(rename_all = "camelCase", default)]
140pub struct Settings {
141    /// The application theme.
142    #[serde_as(deserialize_as = "DefaultOnError")]
143    pub theme: Theme,
144    /// The keyboard shortcuts.
145    #[serde_as(deserialize_as = "MapSkipError<_, _>")]
146    pub shortcuts: BTreeMap<ShortcutKey, String>,
147    /// The mapping from widget IDs to their respective settings.
148    #[serde_as(deserialize_as = "MapSkipError<_, _>")]
149    pub widgets: BTreeMap<String, WidgetSettings>,
150}

152/// A patch for partial updates to [`Settings`].
153#[derive(Debug, Default, Deserialize, specta::Type)]
154#[serde(rename_all = "camelCase", default)]
155pub struct SettingsPatch {
156    /// If not `None`, update [`Settings::theme`].
157    #[specta(optional, type = Theme)]
158    pub theme: Option<Theme>,
159    /// If not `None`, update [`Settings::shortcuts`].
160    ///
161    /// Non-specified shortcuts will remain unchanged. If a shortcut value is
162    /// `None`, it means removing that shortcut. Otherwise, it means updating
163    /// or adding that shortcut.
164    #[specta(optional, type = BTreeMap<ShortcutKey, Option<String>>)]
165    pub shortcuts: Option<BTreeMap<ShortcutKey, Option<String>>>,
166    /// If not `None`, update [`Settings::widgets`].
167    ///
168    /// Non-specified widgets will remain unchanged. If a widget settings patch
169    /// is `None`, it means leaving that widget settings unchanged. Otherwise,
170    /// it means applying the patch to that widget settings. If the widget ID
171    /// does not exist, a new widget settings will be created with default
172    /// values, and then the patch will be applied to it.
173    #[specta(optional, type = BTreeMap<String, Option<WidgetSettingsPatch>>)]
174    pub widgets: Option<BTreeMap<String, Option<WidgetSettingsPatch>>>,
175}
