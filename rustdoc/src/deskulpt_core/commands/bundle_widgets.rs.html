14/// Bundle widgets.
15///
16/// This command bundles the specified widgets that exist in the catalog. If
17/// `ids` is not provided, all widgets in the catalog are bundled. Failure to
18/// bundle a widget does not result in an error, but is reported back to the
19/// canvas window via the [`RenderWidgetEvent`]. Moreover, failure to emit a
20/// single [`RenderWidgetEvent`] does not prevent other widgets from being
21/// processed; instead, errors are collected and returned as a single error at
22/// the end, if any.
23///
24/// ### Errors
25///
26/// - Error accessing the widgets directory.
27/// - Error emitting [`RenderWidgetEvent`] for one or more widgets.
28#[command]
29#[specta::specta]
30pub async fn bundle_widgets<R: Runtime>(
31    app_handle: AppHandle<R>,
32    ids: Option<Vec<String>>,
33) -> CmdResult<()> {
34    let widgets_dir = app_handle.widgets_dir()?;
35
36    let widgets: Vec<_> = {
37        let catalog = app_handle.get_widget_catalog();
38        match ids {
39            Some(ids) => ids
40                .into_iter()
41                .filter_map(|id| match catalog.0.get(&id) {
42                    Some(Outcome::Ok(config)) => Some((id, config.entry.clone())),
43                    _ => None,
44                })
45                .collect(),
46            None => catalog
47                .0
48                .iter()
49                .filter_map(|(id, config)| match config {
50                    Outcome::Ok(config) => Some((id.clone(), config.entry.clone())),
51                    _ => None,
52                })
53                .collect(),
54        }
55    };
56
57    if widgets.is_empty() {
58        return Ok(());
59    }
60
61    let mut errors = vec![];
62    for (id, entry) in widgets.into_iter() {
63        let code = match WidgetBundlerBuilder::new(widgets_dir.join(&id), entry)
64            .build()
65            .context("Failed to build widget bundler")
66        {
67            Ok(mut bundler) => bundler
68                .bundle()
69                .await
70                .with_context(|| format!("Failed to bundle widget (id={id})"))
71                .map_or_else(|e| Outcome::Err(format!("{e:?}")), Outcome::Ok),
72            Err(e) => Outcome::Err(format!("{e:?}")),
73        };
74
75        let event = RenderWidgetEvent { id: &id, code };
76        if let Err(e) = event.emit_to(&app_handle, DeskulptWindow::Canvas) {
77            errors.push(e.context(format!("Failed to render widget (id={id})")));
78        }
79    }
80
81    if !errors.is_empty() {
82        let message = errors
83            .into_iter()
84            .map(|e| format!("{e}"))
85            .collect::<Vec<_>>()
86            .join("\n");
87        cmdbail!(message);
88    }
89
90    Ok(())
91}
