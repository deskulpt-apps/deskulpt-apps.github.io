15/// Bundle widgets.
16///
17/// This command bundles the specified widgets that exist in the catalog. If
18/// `ids` is not provided, all widgets in the catalog are bundled. Failure to
19/// bundle an individual widget does not prevent other widgets from being
20/// bundled. Instead, the outcome of each bundling operation is collected and
21/// sent to the canvas window via the [`RenderWidgetsEvent`].
22///
23/// ### Errors
24///
25/// - Error accessing the widgets directory.
26/// - Error emitting the [`RenderWidgetsEvent`].
27#[command]
28#[specta::specta]
29pub async fn bundle_widgets<R: Runtime>(
30    app_handle: AppHandle<R>,
31    ids: Option<Vec<String>>,
32) -> CmdResult<()> {
33    let widgets_dir = app_handle.widgets_dir()?;
34
35    let catalog = app_handle.get_widget_catalog().clone();
36    let widgets: Vec<_> = match ids {
37        Some(ids) => ids
38            .into_iter()
39            .filter_map(|id| catalog.0.get(&id).map(|config| (id, config)))
40            .collect(),
41        None => catalog
42            .0
43            .iter()
44            .map(|(id, config)| (id.clone(), config))
45            .collect(),
46    };
47
48    if widgets.is_empty() {
49        return Ok(());
50    }
51
52    let futs = widgets.into_iter().map(|(id, config)| async move {
53        match config {
54            Outcome::Ok(config) => {
55                let report =
56                    match WidgetBundlerBuilder::new(widgets_dir.join(&id), config.entry.clone())
57                        .build()
58                        .context("Failed to build widget bundler")
59                    {
60                        Ok(mut bundler) => bundler
61                            .bundle()
62                            .await
63                            .with_context(|| format!("Failed to bundle widget (id={id})"))
64                            .map_or_else(|e| Outcome::Err(format!("{e:?}")), Outcome::Ok),
65                        Err(e) => Outcome::Err(format!("{e:?}")),
66                    };
67                (id, report)
68            },
69            Outcome::Err(e) => (id, Outcome::Err(e.clone())),
70        }
71    });
72
73    let reports = futures::future::join_all(futs)
74        .await
75        .into_iter()
76        .collect::<HashMap<_, _>>();
77
78    RenderWidgetsEvent(reports).emit_to(&app_handle, DeskulptWindow::Canvas)?;
79    Ok(())
80}
