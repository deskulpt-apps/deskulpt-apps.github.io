6/// Open the widgets directory or a specific widget directory.
7///
8/// If the widget ID is provided, a specific widget directory will be opened.
9/// Otherwise, the widgets directory will be opened.
10///
11/// ### Errors
12///
13/// - Widget ID is provided but does not exist in the collection.
14/// - Failed to access the widgets directory.
15/// - Error opening the directory.
16#[command]
17#[specta::specta]
18pub async fn open_widget<R: Runtime>(
19    app_handle: AppHandle<R>,
20    id: Option<String>,
21) -> SerResult<()> {
22    let widgets_dir = app_handle.widgets_dir()?;
23
24    if let Some(id) = id {
25        let widget_dir = app_handle.widget_dir(&id)?;
26        open::that_detached(widget_dir)?;
27    } else {
28        open::that_detached(widgets_dir)?;
29    };
30
31    Ok(())
32}
