3use serde::Serialize;

5/// A serializable error type for the commands.
6#[derive(thiserror::Error, Debug)]
7pub enum CmdError {
8    #[error("{0:?}")]
9    Anyhow(#[from] anyhow::Error),
10    #[error("I/O error: {0}")]
11    Io(#[from] std::io::Error),
12    #[error("Global shortcut error: {0}")]
13    TauriPluginGlobalShortcut(#[from] tauri_plugin_global_shortcut::Error),
14}
15
16impl Serialize for CmdError {
17    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
18    where
19        S: serde::ser::Serializer,
20    {
21        serializer.serialize_str(self.to_string().as_ref())
22    }
23}
24
25impl specta::Type for CmdError {
26    fn inline(
27        type_map: &mut specta::TypeCollection,
28        generics: specta::Generics,
29    ) -> specta::datatype::DataType {
30        <String as specta::Type>::inline(type_map, generics)
31    }
32
33    fn reference(
34        type_map: &mut specta::TypeCollection,
35        generics: &[specta::datatype::DataType],
36    ) -> specta::datatype::reference::Reference {
37        <String as specta::Type>::reference(type_map, generics)
38    }
39}
40
41/// The return type of all Deskulpt core commands.
42pub type CmdResult<T> = Result<T, CmdError>;
43
44/// [anyhow::anyhow!] that returns a [CmdError].
45macro_rules! cmderr {
46    ($msg:literal $(,)?) => {
47        $crate::commands::error::CmdError::from(anyhow::anyhow!($msg))
48    };
49    ($err:expr $(,)?) => {
50        $crate::commands::error::CmdError::from(anyhow::anyhow!($err))
51    };
52    ($fmt:expr, $($arg:tt)*) => {
53        $crate::commands::error::CmdError::from(anyhow::anyhow!($fmt, $($arg)*))
54    };
55}
56
57/// [anyhow::bail!] that early returns a [CmdError].
58macro_rules! cmdbail {
59    ($msg:literal $(,)?) => {
60        return Err($crate::commands::error::cmderr!($msg))
61    };
62    ($err:expr $(,)?) => {{
63        return Err($crate::commands::error::cmderr!($err))
64    }};
65    ($fmt:expr, $($arg:tt)*) => {
66        return Err($crate::commands::error::cmderr!($fmt, $($arg)*))
67    };
68}
69
70pub(super) use {cmdbail, cmderr};
