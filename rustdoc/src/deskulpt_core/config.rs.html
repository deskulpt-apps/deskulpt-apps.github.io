16/// The Deskulpt widget manifest.
17#[derive(Debug, Deserialize)]
18#[serde(rename_all = "camelCase")]
19pub struct WidgetManifest {
20    /// The name of the widget.
21    ///
22    /// This is purely used for display purposes. It does not need to be related
23    /// to the widget directory name, and it does not need to be unique.
24    pub name: String,
25    /// The entry point of the widget.
26    ///
27    /// This is the path to the file that exports the widget component. The path
28    /// should be relative to the widget directory.
29    pub entry: String,
30    /// Whether to ignore the widget.
31    ///
32    /// If set to true, the widget will not be discovered by the application.
33    /// This is useful for temporarily disabling a widget without removing it.
34    #[serde(default, skip_serializing)]
35    ignore: bool,
36}

38/// The Node.js package manifest.
39#[derive(Debug, Default, Deserialize)]
40#[serde(rename_all = "camelCase")]
41pub struct PackageManifest {
42    #[serde(default)]
43    pub dependencies: HashMap<String, String>,
44}
45
46/// Helper trait for loading manifest files from a directory.
47trait LoadManifest: Sized + DeserializeOwned {
48    /// The name of the manifest file.
49    const FILE_NAME: &'static str;
50
51    /// Load the manifest file from the given directory.
52    ///
53    /// Specially, this method returns `Ok(None)` if the target file does not
54    /// exist and does not treat it as an error.
55    fn load(dir: &Path) -> Result<Option<Self>> {
56        let path = dir.join(Self::FILE_NAME);
57        if !path.exists() {
58            return Ok(None);
59        }
60        let file = File::open(path)?;
61        let reader = BufReader::new(file);
62        let config = serde_json::from_reader(reader)?;
63        Ok(Some(config))
64    }
65}
66
67impl LoadManifest for WidgetManifest {
68    const FILE_NAME: &'static str = "deskulpt.widget.json";
69}
70
71impl LoadManifest for PackageManifest {
72    const FILE_NAME: &'static str = "package.json";
73}

75/// Full configuration of a Deskulpt widget.
76#[derive(Debug, Serialize, Deserialize, specta::Type)]
77#[serde(rename_all = "camelCase")]
78pub struct WidgetConfig {
79    /// The name of the widget.
80    pub name: String,
81    /// The entry point of the widget.
82    pub entry: String,
83    /// The dependencies of the widget.
84    pub dependencies: HashMap<String, String>,
85}
86
87impl WidgetConfig {
88    /// Read full widget configuration from a directory.
89    ///
90    /// Specially, this method returns `Ok(None)` if the directory does not
91    /// contain a widget configuration file or if the widget is explicitly
92    /// marked as ignored in the configuration file.
93    pub fn load(dir: &Path) -> Result<Option<Self>> {
94        let widget_manifest = match WidgetManifest::load(dir)
95            .with_context(|| format!("Failed to load {}", WidgetManifest::FILE_NAME))?
96        {
97            Some(widget_manifest) if !widget_manifest.ignore => widget_manifest,
98            _ => return Ok(None),
99        };
100
101        let package_manifest = PackageManifest::load(dir)
102            .with_context(|| format!("Failed to load {}", PackageManifest::FILE_NAME))?
103            .unwrap_or_default();
104
105        Ok(Some(WidgetConfig {
106            name: widget_manifest.name,
107            entry: widget_manifest.entry,
108            dependencies: package_manifest.dependencies,
109        }))
110    }
111}

113/// The widget catalog.
114///
115/// This is a collection of all widgets discovered locally, mapped from their
116/// widget IDs to their configurations. Invalid widgets are also included with
117/// their error messages.
118#[derive(Debug, Default, Serialize, Deserialize, specta::Type)]
119pub struct WidgetCatalog(pub BTreeMap<String, Outcome<WidgetConfig>>);
120
121impl WidgetCatalog {
122    /// Load the widget catalog from the given directory.
123    ///
124    /// This scans all top-level subdirectories and attempts to load them as
125    /// widgets. Non-widget directories are simply ignored. See
126    /// [`WidgetConfig::load`] for more details.
127    pub fn load(dir: &Path) -> Result<Self> {
128        let mut catalog = Self::default();
129
130        let entries = std::fs::read_dir(dir)?;
131        for entry in entries {
132            let entry = entry?;
133
134            let path = entry.path();
135            if !path.is_dir() {
136                continue; // Non-directory entries are not widgets, skip
137            }
138
139            if let Some(config) = WidgetConfig::load(&path)
140                .map(|opt| opt.map(Outcome::Ok))
141                .unwrap_or_else(|e| Some(Outcome::Err(format!("{e:?}"))))
142            {
143                // Since each widget must be at the top level of the widgets
144                // directory, the directory names must be unique and we can use
145                // them as widget IDs
146                let id = entry.file_name().to_string_lossy().to_string();
147                catalog.0.insert(id, config);
148            }
149        }
150
151        Ok(catalog)
152    }
153
154    /// Compute a settings patch to synchronize with the catalog.
155    ///
156    /// This method compares the given widget settings with catalog and
157    /// generates a patch such that:
158    ///
159    /// - If a widget exists in the settings but not in the catalog, it will be
160    ///   removed from the settings.
161    /// - If a widget exists in the catalog but not in the settings, it will be
162    ///   added to the settings with an empty patch, which results in default
163    ///   settings.
164    /// - If a widget exists in both, no changes are made.
165    pub fn compute_settings_patch(&self, settings: &Settings) -> SettingsPatch {
166        let mut patches = BTreeMap::new();
167
168        for e in itertools::merge_join_by(
169            settings.widgets.iter(), // settings (to be synced)
170            self.0.iter(),           // catalog (truth)
171            |(a, _), (b, _)| a.cmp(b),
172        ) {
173            match e {
174                itertools::EitherOrBoth::Left((id, _)) => {
175                    patches.insert(id.clone(), None);
176                },
177                itertools::EitherOrBoth::Right((id, _)) => {
178                    patches.insert(id.clone(), Some(Default::default()));
179                },
180                itertools::EitherOrBoth::Both(_, _) => {},
181            }
182        }
183
184        SettingsPatch {
185            widgets: Some(patches),
186            ..Default::default()
187        }
188    }
189}
